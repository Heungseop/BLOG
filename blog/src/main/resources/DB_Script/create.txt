

create table member (
         no         number      constraint user_no primary key
       , id		    varchar2(60)
       , nick	    varchar2(10)
       , pw		    varchar2(30)
       , auth	    varchar2(10)
       , reg_date	date
       , reg_user   varchar2(60)
       , mod_date   date
       , mod_user   varchar2(60)
       )
       ;
create table board(
          no		number          constraint board_no primary key
        , board_kind varchar2(10)
        , title		varchar2(100)
        , content	varchar2(4000)
        , read		number
        , like_cnt	number
        , reg_date	date
        , reg_user   varchar2(60)
        , mod_date   date
        , mod_user   varchar2(60)
)
  ;
create table reply(
          board_no		number 
        , no			number      
        , content		varchar2(1000)
        , like_cnt		number
        , reg_date		date
        , reg_user		varchar2(60)
        , mod_date		date
        , mod_user	varchar2(60)
        , constraint reply_pk primary key (board_no, no)
);

create table common_code(
       grp_cd   varchar2(100)
     , cd     varchar2(20)
     , nm     varchar2(100)
     , str      varchar2(100)
--     , ord_no	number
     , reg_date	 date
     , reg_user   varchar2(60)
     , mod_date   date
     , mod_user   varchar2(60)
     , constraint common_code_pk primary key (grp_cd, nm)
     )
;

create table grp_code(
       grp_cd   	varchar2(10)
     , nm     		varchar2(20)
     , reg_date	 	date
     , reg_user   	varchar2(60)
     , mod_date   	date
     , mod_user   	varchar2(60)
     , constraint 	grp_code_pk primary key (grp_cd)
     )
;
